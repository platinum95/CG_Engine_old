
//== SHADER COMPILE STATUS = TRUE
#version 330

layout (location = 0) in vec3 vPosition;
layout (location = 1) in vec4 fColor;
uniform float time;
out vec4 PassColour;
#define SCALE_FACTOR 0.5
mat4 scale = mat4(  1/time, 0, 0, 0,
					0, 1/time, 0 ,0,
					0, 0, SCALE_FACTOR, 0,
					0, 0, 0, 1   );

void main(){
	PassColour = fColor;
    gl_Position = scale * vec4(vPosition.x, vPosition.y, vPosition.z, 1.0);
}
